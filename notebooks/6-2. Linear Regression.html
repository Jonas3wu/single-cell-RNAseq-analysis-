from scipy.stats import linregress, pearsonr, spearmanr

# Extract vectors
x = expr["CD33"].values
y = expr["IL1RAP"].values

# Linear regression

# Correlations
r_pearson, p_pearson = pearsonr(x, y)
r_spearman, p_spearman = spearmanr(x, y)

# --- plot ---
plt.figure(figsize=(5, 5))
plt.scatter(x, y, s=8, alpha=0.4)

# regression line
xx = np.linspace(x.min(), x.max(), 200)
yy = lr.intercept + lr.slope * xx
plt.plot(xx, yy, linewidth=2)

# two distintive cell population exclusively express either CD33 or IL1RAP, 
#let's find out who are they - define expression-based group
import numpy as np
import pandas as pd

# thresholds (log1p scale)
cd33_thr = 0.25
il1rap_thr = 0.25

expr["CD33_pos"] = expr["CD33"] > cd33_thr
expr["IL1RAP_pos"] = expr["IL1RAP"] > il1rap_thr

expr["expr_group"] = np.select(
    [
        expr["CD33_pos"] & ~expr["IL1RAP_pos"],
        ~expr["CD33_pos"] & expr["IL1RAP_pos"],
        expr["CD33_pos"] & expr["IL1RAP_pos"],
    ],
    [
        "CD33+ IL1RAP",
        "CD33 IL1RAP+",
        "CD33+ IL1RAP+",
    ],
    default="Double negative"
)
